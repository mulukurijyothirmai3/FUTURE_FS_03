<script>
let count = 0;
let total = 0;

function addItem(price) {
    count++;
    total += price;
    updateCart();
}

function removeItem(price) {
    if(count > 0){
        count--;
        total -= price;
        if(total < 0) total = 0;
        updateCart();
    }
}

function updateCart() {
    document.getElementById("count").innerText = count;
    document.getElementById("total").innerText = total;
}

function reserveTable(event) {
    event.preventDefault();
    alert("Table Reserved Successfully! 🍽️");
}
